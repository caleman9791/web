Para modificar la URL visible en la barra de direcciones del navegador sin recargar la página, puedes utilizar la API del historial (history.pushState() y history.replaceState()) en conjunto con la manipulación del evento popstate. Aquí tienes un ejemplo básico:
<!DOCTYPE html>
<html>
<head>
    <!-- Tu código del encabezado aquí -->
</head>
<body>
    <header>
        <!-- Tu encabezado aquí -->
    </header>

    <nav>
        <a href="/home" id="homeLink">Inicio</a>
        <a href="/about" id="aboutLink">Acerca de</a>
        <a href="/contact" id="contactLink">Contacto</a>
    </nav>

    <main id="content">
        <!-- Contenedor para cargar contenido dinámicamente -->
    </main>

    <footer>
        <!-- Tu pie de página aquí -->
    </footer>

    <!-- Tu código de scripts JavaScript aquí -->
    <script>
        // Funciones para cargar contenido en base a las URLs simuladas
        function loadContent(url) {
            // Aquí cargas el contenido correspondiente a la URL
            // Puedes usar AJAX, fetch, u otros métodos para obtener el contenido
        }

        // Función para manipular el evento 'popstate'
        function handlePopState(event) {
            // Obtener la URL actual
            var url = location.pathname;

            // Cargar el contenido correspondiente a la URL
            loadContent(url);
        }

        // Asociar funciones a los enlaces correspondientes
        document.getElementById('homeLink').addEventListener('click', function(event) {
            event.preventDefault();
            var url = '/home';
            // Modificar la URL visible en la barra de direcciones sin recargar la página
            history.pushState(null, null, url);
            // Cargar el contenido correspondiente a la URL
            loadContent(url);
        });

        document.getElementById('aboutLink').addEventListener('click', function(event) {
            event.preventDefault();
            var url = '/about';
            history.pushState(null, null, url);
            loadContent(url);
        });

        document.getElementById('contactLink').addEventListener('click', function(event) {
            event.preventDefault();
            var url = '/contact';
            history.pushState(null, null, url);
            loadContent(url);
        });

        // Manipular el evento 'popstate' para cargar el contenido correspondiente a la URL al retroceder o avanzar en el historial
        window.addEventListener('popstate', handlePopState);
    </script>
</body>
</html>

function loadContent(url) {
    fetch(url)
        .then(response => response.text())
        .then(data => {
            document.getElementById('content').innerHTML = data;
        })
        .catch(error => {
            console.error('Error al cargar el contenido:', error);
        });
}
